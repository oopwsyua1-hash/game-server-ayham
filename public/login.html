async function login() { // ... الكود تبعك const res = await fetch(`${API_URL}/api/login`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ email, password }) }); const data = await res.json(); if (data.token) { // هون حط السطرين localStorage.setItem('token', data.token); window.location.href = '/me.html'; } else { alert(data.error || 'خطأ بالتسجيل'); } }
